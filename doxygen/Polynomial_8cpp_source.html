17 template <typename PolyT, typename MonomialT>
18 FailureOr<PolyT> fromMonomialsImpl(ArrayRef<MonomialT> monomials) {
19  // A polynomial's terms are canonically stored in order of increasing degree.
20  auto monomialsCopy = llvm::SmallVector<MonomialT>(monomials);
21  std::sort(monomialsCopy.begin(), monomialsCopy.end());
22 
23  // Ensure non-unique exponents are not present. Since we sorted the list by
24  // exponent, a linear scan of adjancent monomials suffices.
25  if (std::adjacent_find(monomialsCopy.begin(), monomialsCopy.end(),
26  [](const MonomialT &lhs, const MonomialT &rhs) {
27  return lhs.getExponent() == rhs.getExponent();
28  }) != monomialsCopy.end()) {
29  return failure();
30  }
31 
32  return PolyT(monomialsCopy);
33 }

45 template <typename PolyT, typename MonomialT, typename CoeffT>
46 PolyT fromCoefficientsImpl(ArrayRef<CoeffT> coeffs) {
47  llvm::SmallVector<MonomialT> monomials;
48  auto size = coeffs.size();
49  monomials.reserve(size);
50  for (size_t i = 0; i < size; i++) {
51  monomials.emplace_back(coeffs[i], i);
52  }
53  auto result = PolyT::fromMonomials(monomials);
54  // Construction guarantees unique exponents, so the failure mode of
55  // fromMonomials can be bypassed.
56  assert(succeeded(result));
57  return result.value();
58 }
